    				 //Disable backbutton
						function preventBack() { window.history.forward(); }
						    setTimeout("preventBack()", 0);
						    window.onunload = function () { null };
    				
    			
    				  // Disable Mouse Right click
    document.addEventListener("contextmenu", function(event){
	    event.preventDefault();
	    alert('mouse right click is disabled');    
    }, false);
    				
    			
    				// Disable Copy and Paste
    document.onselectstart = () => {
      event.preventDefault();
      alert('Disabled copy Paste');
      }
      
    				
    			
    				// Disable keys on keyboard
     document.onkeydown = function(e) {
        if(event.keyCode == 123) {
          return false;
        }
        if(e.ctrlKey && e.shiftKey && e.keyCode == 'I'.charCodeAt(0)) {
        	alert('Disabled Web Devloper Inspector(Ctrl + Shift + I)');
          return false;
        }
        if(e.ctrlKey && e.shiftKey && e.keyCode == 'C'.charCodeAt(0)) {
        	alert('Disabled Copy option(Ctrl + Shift + C)');
          return false;
        }
        if(e.ctrlKey && e.shiftKey && e.keyCode == 'J'.charCodeAt(0)) {
        	alert('Disabled page console option(Ctrl + Shift + J)');
          return false;
        }
        if(e.ctrlKey && e.keyCode == 'U'.charCodeAt(0)) {
        	alert('Disabled entering unicode character(Ctrl + Shift + U)');
          return false;
        }
        if (e.ctrlKey && e.key == 'p') {
            alert('Disabled Print Page action (Ctrl + p)');
            e.cancelBubble = true;
            e.preventDefault();
            e.stopImmediatePropagation();
        }
      }
    				
    			
    				  /** TO DISABLE PRINTS WHIT CTRL+P **/
      document.addEventListener('keydown', (e) => {
          if (e.ctrlKey && e.key == 'p') {
              alert('Disabled Print Page action (Ctrl + p)');
              e.cancelBubble = true;
              e.preventDefault();
              e.stopImmediatePropagation();
              return false;
          }
      });
